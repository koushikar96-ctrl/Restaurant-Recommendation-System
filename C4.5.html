import numpy as np
import pandas as pd
from sklearn.tree import DecisionTreeClassifier, plot_tree
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score
import matplotlib.pyplot as plt
from sklearn.preprocessing import LabelEncoder  # Import LabelEncoder

def calculate_entropy(data, target_column):
    """Calculate the entropy of the target column."""
    class_counts = data[target_column].value_counts()
    probabilities = class_counts / len(data)
    entropy = -np.sum(probabilities * np.log2(probabilities))
    return entropy

def calculate_information_gain(data, attribute_column, target_column):
    """Calculate information gain of an attribute given a target column."""
    original_entropy = calculate_entropy(data, target_column)
    attribute_values = data[attribute_column].unique()
    weighted_entropy = 0
    for value in attribute_values:
        subset = data[data[attribute_column] == value]
        subset_entropy = calculate_entropy(subset, target_column)
        weight = len(subset) / len(data)
        weighted_entropy += weight * subset_entropy
    information_gain = original_entropy - weighted_entropy
    return information_gain

def calculate_split_information(data, attribute_column):
    """Calculate the Split Information for a given attribute (used in Gain Ratio)."""
    attribute_values = data[attribute_column].value_counts()
    total = len(data)
    split_info = -np.sum((attribute_values / total) * np.log2(attribute_values / total))
    return split_info

def calculate_gain_ratio(data, attribute_column, target_column):
    """Calculate Gain Ratio for a feature."""
    information_gain = calculate_information_gain(data, attribute_column, target_column)
    split_information = calculate_split_information(data, attribute_column)
    if split_information == 0:  # Avoid division by zero
        return 0
    return information_gain / split_information

def best_feature_gain_ratio(data, target_column):
    """Find the feature with the highest Gain Ratio."""
    features = data.select_dtypes(include=['object', 'category', 'bool']).columns.tolist()
    if target_column in features:
        features.remove(target_column)
    gain_ratios = {feature: calculate_gain_ratio(data, feature, target_column) for feature in features}
    best = max(gain_ratios, key=gain_ratios.get)
    return best, gain_ratios[best]

def build_decision_tree_c45(data, target_column):
    """Build the decision tree using C4.5 algorithm."""
    # Prepare data for the classifier
    X = data.drop(columns=["rate", target_column])  # Adjust this as per dataset
    y = data[target_column]

    # Convert string/object columns to numerical using Label Encoding
    for col in X.select_dtypes(include=['object']).columns:
        le = LabelEncoder()
        X[col] = le.fit_transform(X[col])  # Convert categorical data into numbers

    # Split data into training and test sets
    X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.3, random_state=42)

    # Build the decision tree classifier using Gain Ratio (C4.5)
    clf = DecisionTreeClassifier(criterion='entropy', random_state=42, max_depth=4)  # C4.5 uses entropy internally
    clf.fit(X_train, y_train)

    # Predict the results
    y_pred = clf.predict(X_test)

    # Calculate performance measures
    accuracy = accuracy_score(y_test, y_pred)
    precision = precision_score(y_test, y_pred, average='weighted', zero_division=0)
    recall = recall_score(y_test, y_pred, average='weighted', zero_division=0)
    f1 = f1_score(y_test, y_pred, average='weighted', zero_division=0)

    return clf, accuracy, precision, recall, f1

def calculate_gain_ratio_for_categorical_features(data, target_column):
    """Calculate the Gain Ratio for each categorical feature in the dataset."""
    categorical_features = data.select_dtypes(include=['object', 'category']).columns.tolist()
    
    # Remove the target column if it is included in the categorical features
    if target_column in categorical_features:
        categorical_features.remove(target_column)
    
    gain_ratio_values = {}
    for feature in categorical_features:
        gain_ratio = calculate_gain_ratio(data, feature, target_column)
        gain_ratio_values[feature] = gain_ratio
    
    return gain_ratio_values

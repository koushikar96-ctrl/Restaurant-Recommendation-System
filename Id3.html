import pandas as pd
import numpy as np
from sklearn.tree import DecisionTreeClassifier, export_graphviz
from sklearn.preprocessing import LabelEncoder
from sklearn.model_selection import train_test_split
from sklearn.metrics import accuracy_score, precision_score, recall_score, f1_score
import graphviz
from IPython.display import Image, display

def calculate_entropy(y):
    """Calculate entropy of the target variable."""
    value_counts = y.value_counts(normalize=True)
    return -np.sum(value_counts * np.log2(value_counts + 1e-9))  # Adding epsilon to avoid log(0)

def calculate_information_gain(X, y, feature):
    """Calculate Information Gain for a specific feature."""
    total_entropy = calculate_entropy(y)
    
    # Calculate weighted entropy for each subset
    feature_values = X[feature].unique()
    weighted_entropy = 0
    for value in feature_values:
        subset = y[X[feature] == value]
        weighted_entropy += (len(subset) / len(y)) * calculate_entropy(subset)

    information_gain = total_entropy - weighted_entropy
    return information_gain

def preprocess_data(df):
    # Encode categorical features
    le = LabelEncoder()
    df['online_order'] = le.fit_transform(df['online_order'])
    df['book_table'] = le.fit_transform(df['book_table'])
    df['listed_in(type)'] = le.fit_transform(df['listed_in(type)'])
    df['diet_type'] = le.fit_transform(df['diet_type'])
    df['location'] = le.fit_transform(df['location'])

    # Convert the rate into a categorical target variable
    df['target'] = pd.cut(df['rate'], bins=[0, 3, 4, 5], labels=['low', 'medium', 'high'])
    return df

def split_data(df):
    X = df[['online_order', 'book_table', 'votes', 'approx_cost', 'listed_in(type)', 'diet_type', 'location']]
    y = df['target']
    return train_test_split(X, y, test_size=0.3, random_state=42)

def train_decision_tree(X_train, y_train):
    # Set max_depth to limit the size of the tree
    dt_classifier = DecisionTreeClassifier(criterion='entropy', max_depth=3, min_samples_split=2, min_samples_leaf=1, random_state=42)
    dt_classifier.fit(X_train, y_train)
    return dt_classifier

def visualize_tree(classifier, feature_names):
    dot_data = export_graphviz(
        classifier, 
        out_file=None, 
        feature_names=feature_names,  
        class_names=['low', 'medium', 'high'],  
        filled=True, 
        rounded=True,  
        special_characters=True
    )  
    
    # Render the graph to PNG format
    graph = graphviz.Source(dot_data)
    display(Image(graph.pipe(format='png')))

def evaluate_model(y_test, y_pred):
    accuracy = accuracy_score(y_test, y_pred)
    precision = precision_score(y_test, y_pred, average='weighted')
    recall = recall_score(y_test, y_pred, average='weighted')
    f1 = f1_score(y_test, y_pred, average='weighted')

    print(f"Accuracy: {accuracy:.4f}")
    print(f"Precision: {precision:.4f}")
    print(f"Recall: {recall:.4f}")
    print(f"F1 Score: {f1:.4f}")

def main():
    # Load dataset
    df = pd.read_csv('Zomato-restaurant-data.csv', encoding='latin1')  # Adjust the filename and encoding as needed
    df = df.loc[:, ~df.columns.str.contains('^Unnamed')]  # Remove unnamed columns
    

    # Preprocess the data
    df = preprocess_data(df)

    # Calculate and print entropy for the entire dataset
    total_entropy = calculate_entropy(df['target'])
    print(f"Total Entropy of the dataset: {total_entropy:.4f}")

    # Calculate Entropy and Information Gain for each categorical attribute
    information_gains = {}
    features = ['online_order', 'book_table', 'votes', 'approx_cost', 'listed_in(type)', 'diet_type', 'location']
    
    for feature in features:
        ig = calculate_information_gain(df, df['target'], feature)
        information_gains[feature] = ig
        print(f"Information Gain for {feature}: {ig:.4f}")

    # Split the data
    X_train, X_test, y_train, y_test = split_data(df)

    # Train the Decision Tree Classifier
    dt_classifier = train_decision_tree(X_train, y_train)

    # Visualize the Decision Treefor 
    print("\t\t\tID3 Decision Tree classifier")
    visualize_tree(dt_classifier, X_train.columns)

    # Make predictions on the test set
    y_pred = dt_classifier.predict(X_test)

    # Evaluate the model
    evaluate_model(y_test, y_pred)

# Run the main function
if __name__ == "__main__":
    main()
